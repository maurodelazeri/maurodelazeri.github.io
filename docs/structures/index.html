#Trades

#TradeRequest

Interaction between the client and a trade server for executing the order placing operation is performed by using trade requests. The trade request is represented by the special predefined protobuf message of TradeRequest type, which contain all the fields necessary to perform trade deals. TradeRequest is used by Buy(), Sell(), BuyLimit(), SellLimit(), BuyStop(), SellStop(). Each function uses specific fields and if not specified it uses the default value specified on Trades Documentation. Any extra field specified that is not necessary is ignored.

message TradeRequest {
    string                         token            = 1;  // Token to authenticate
    bool                           asynchronous     = 2;  // Asynchronous execution
    string                         venue            = 3;  // Trade Venue
    string                         product          = 4;  // Trade Product
    double                         volume           = 5;  // Requested volume for a deal in lots
    double                         price            = 6;  // Price
    double                         sl               = 7;  // Stop Loss level of the order
    double                         tp               = 8;  // Take Profit level of the order
    double                         deviation        = 9;  // Maximal possible deviation from the requested price
    ENUM_ORDER_TYPE                type             = 10; // Order type
    ENUM_ORDER_TYPE_FILLING        type_filling     = 11; // Order execution type
    ENUM_ORDER_TYPE_TIME           type_time        = 12; // Order expiration type
    int64                          expiration       = 13; // Order expiration time (for the orders of ORDER_TIME_SPECIFIED type)
    string                         comment          = 14; // Order comment
};

#TradeResponse

As result of a trade request, a trade server returns data about the trade request processing result as a special predefined protobuf message of TradeResponse type.

message TradeResponse {
    ENUM_TRADE_RETCODE             retcode                   = 1;  // Operation return code
    string                         deal_id                   = 2;  // Deal id, if it is performed
    string                         order_id                  = 3;  // Order id, if it is placed
    double                         volume                    = 4;  // Deal volume, confirmed by broker
    double                         price                     = 5;  // Deal price, confirmed by broker
    double                         bid                       = 6;  // Current Bid price
    double                         ask                       = 7;  // Current Ask price
    string                         comment                   = 8;  // Broker comment to operation (by default it is filled by description of trade server return code)
    string                         request_id                = 9;  // Request ID set by the client during the dispatch
    string                         retcode_external          = 10; // Return code of an external trading system
};

#Orders

#Order

This represent the structure of an order

message Order {
    string                              order_id                        = 1;  // Order id. Unique uuid assigned to each order
    int64                               time_setup                      = 2;  // Order setup time
    ENUM_ORDER_TYPE                     type                            = 3;  // Order type
    ENUM_ORDER_STATE                    state                           = 4;  // Order state
    int64                               expiration                      = 5;  // Order expiration time
    int64                               time_done                       = 6;  // Order execution or cancellation time
    ENUM_ORDER_TYPE_FILLING             type_filling                    = 7;  // Order filling type
    ENUM_ORDER_TYPE_TIME                lifetime                        = 8;  // Order lifetime
    ENUM_ORDER_REASON                   reason                          = 9;  // The reason or source for placing an order
    string                              position_id                     = 10; // Position identifier that is set to an order as soon as it is executed. Each executed order results in a deal that opens or modifies an already existing position. The identifier of exactly this position is set to the executed order at this moment.
    string                              position_by_id                  = 11; // Identifier of an opposite position used for closing by order  ORDER_TYPE_CLOSE_BY
    double                              initial_volume                  = 12; // Order initial volume
    double                              current_volume                  = 13; // Order current volume
    double                              price_open                      = 14; // Price specified in the order
    double                              sl                              = 15; // Stop Loss value
    double                              tp                              = 16; // Take Profit value
    double                              price_current                   = 17; // The current price of the order
    double                              price_stop_limit                = 18; // The Limit order price for the StopLimit order
    string                              product                         = 19; // Product of the order
    string                              venue                           = 20; // Venue of the order
    string                              comment                         = 21; // Order comment
    string                              external_id                     = 22; // Order identifier in an external trading system (on the Venue)
};

message Position {
    string                               position_id                     = 1;  // Position id. Unique uuid assigned to each newly opened position. It usually matches the id of an order used to open the position except when the id is changed as a result of service operations on the server, for example, when charging swaps with position re-opening. To find an order used to open a position, apply the POSITION_IDENTIFIER property.
    int64                                open_time                       = 2;  // Position open time
    int64                                update_time                     = 3;  // Position changing time
    string                               identifier                      = 4;  // Position identifier is a unique number assigned to each re-opened position. It does not change throughout its life cycle and corresponds to the ticket of an order used to open a position. Check the docs for more info...
    ENUM_POSITION_TYPE                   type                            = 5;  // Position type
    ENUM_POSITION_REASON                 reason                          = 6;  // The reason for opening a position
    double                               volume                          = 7;  // Position volume
    double                               price_open                      = 8;  // Position open price
    double                               sp                              = 9;  // Stop Loss level of opened position
    double                               tp                              = 10; // Take Profit level of opened position
    double                               current_price                   = 11; // Current price of the position
    double                               swap                            = 12; // Cumulative swap
    double                               profit                          = 13; // Current profit
    string                               product                         = 14; // Deal product
    string                               venue                           = 15; // Deal venue
    string                               external_id                     = 16; // Deal identifier in an external trading system (on the Exchange
};

#TradeTransaction

When performing some definite actions on a trade account, its state changes and sends events about the changes using TradeTransaction structure

message TradeTransaction {
    string                        deal_id          = 1; // Deal id
    string                        order_id         = 2; // Order id
    string                        venue            = 3; // Trade venue name
    string                        product          = 4; // Trade product name
    ENUM_TRADE_TRANSACTION_TYPE   type             = 5; // Trade transaction type
    ENUM_ORDER_TYPE               order_type       = 6; // Order type
    ENUM_ORDER_STATE              order_state      = 7; // Order state
    ENUM_DEAL_TYPE                deal_type        = 8; // Deal type
    ENUM_ORDER_TYPE_TIME          time_type        = 9; // Order type by action period
    int64                         time_expiration  = 10; // Order expiration time
    double                        price            = 11; // Price
    double                        price_trigger    = 12; // Stop limit order activation price
    double                        price_sl         = 13; // Stop Loss level
    double                        price_tp         = 14; // Take Profit level
    double                        volume           = 15; // Volume
    string                        position_id      = 16; // Position id
    string                        position_by_id   = 17; // Position id of an opposite position
};
